    const caixaEletronico = {
    notas: {
      100: 200,
      50: 200,
      20: 200,
      10: 200,
      5: 200,
      2: 1000
    },

    function calcularNotas(saque) {
    const notasDisponiveis = [100, 50, 20, 10, 5, 2];
    const notasContagem = {};

    for (const nota of notasDisponiveis) {
        const qtdNotas = Math.floor(saque / nota);
        if (qtdNotas > 0) {
            notasContagem[nota] = qtdNotas;
            saque -= qtdNotas * nota;
        }
      }
    alert("Quantidade de notas: " + JSON.stringify(notasContagem));
    }

    function realizarSaque() {
    const valorSaque = parseFloat(prompt("Digite o valor do saque: R$"));

    if (isNaN(valorSaque) || valorSaque <= 0) {
        alert("Valor inválido. O saque deve ser um número maior que zero.");
        return;
    }

    const notasSaque = calcularNotas(valorSaque);

    if (Object.values(notasSaque).every(qtd => qtd === 0)) {
        alert("Não é possível fornecer o saque desejado com as notas disponíveis.");
    } else {
        alert(`\nSaque de R$ ${valorSaque.toFixed(2)} com as seguintes notas:`);
        for (const [nota, quantidade] of Object.entries(notasSaque)) {
            alert(`R$ ${parseFloat(nota).toFixed(2)}: ${quantidade} nota(s)`);
        }
      }
    }
